174 uint32_t sdram_init (void)
175 {
176  uint32_t pclk, temp;
177  uint64_t tmpclk;
178 
179  pinConfig(); //Full 32-bit Data bus, 24-bit Address
180 
181  /* Select EMC clock-out */
182  LPC_SCU->SFSCLK_0 = MD_PLN_FAST;
183  LPC_SCU->SFSCLK_1 = MD_PLN_FAST;
184  LPC_SCU->SFSCLK_2 = MD_PLN_FAST;
185  LPC_SCU->SFSCLK_3 = MD_PLN_FAST;
186 
187  LPC_EMC->CONTROL = 0x00000001;
188  LPC_EMC->CONFIG = 0x00000000;
189  LPC_EMC->DYNAMICCONFIG0 = 1<<14 | 0<<12 | 2<<9 | 1<<7; /* 256Mb, 8Mx32, 4 banks, row=12, column=9 */
190 
191  pclk = CGU_GetPCLKFrequency(CGU_PERIPHERAL_M4CORE);
192 
193  LPC_EMC->DYNAMICRASCAS0 = 0x00000202; /* 2 RAS, 2 CAS latency */
194  LPC_EMC->DYNAMICREADCONFIG = 0x00000001; /* Command delayed strategy, using EMCCLKDELAY */
195 
196  LPC_EMC->DYNAMICRP = NS2CLK(pclk, 20);
197  LPC_EMC->DYNAMICRAS = NS2CLK(pclk, 42);
198  LPC_EMC->DYNAMICSREX = NS2CLK(pclk, 63);
199  LPC_EMC->DYNAMICAPR = 0x00000005;
200  LPC_EMC->DYNAMICDAL = 0x00000005;
201  LPC_EMC->DYNAMICWR = 2;
202  LPC_EMC->DYNAMICRC = NS2CLK(pclk, 63);
203  LPC_EMC->DYNAMICRFC = NS2CLK(pclk, 63);
204  LPC_EMC->DYNAMICXSR = NS2CLK(pclk, 63);
205  LPC_EMC->DYNAMICRRD = NS2CLK(pclk, 14);
206  LPC_EMC->DYNAMICMRD = 0x00000002;
207 
208  TIM_Waitus(100); /* wait 100ms */
209  LPC_EMC->DYNAMICCONTROL = 0x00000183; /* Issue NOP command */
210 
211  TIM_Waitus(200); /* wait 200ms */
212  LPC_EMC->DYNAMICCONTROL = 0x00000103; /* Issue PALL command */
213 
214  LPC_EMC->DYNAMICREFRESH = EMC_SDRAM_REFRESH(pclk,70); /* ( n * 16 ) -> 32 clock cycles */
215 
216  //for(i = 0; i < 0x80; i++); /* wait 128 AHB clock cycles */
217  TIM_Waitus(200); /* wait 200ms */
218 
219  tmpclk = (uint64_t)15625*(uint64_t)pclk/1000000000/16;
220  LPC_EMC->DYNAMICREFRESH = tmpclk; /* ( n * 16 ) -> 736 clock cycles -> 15.330uS at 48MHz <= 15.625uS ( 64ms / 4096 row ) */
221 
222  LPC_EMC->DYNAMICCONTROL = 0x00000083; /* Issue MODE command */
223 
224  //Timing for 48/60/72MHZ Bus
225  temp = *((volatile uint32_t *)(SDRAM_BASE | (2<<4| 2)<<(9+2+2))); /* 4 burst, 2 CAS latency */
226  temp = temp;
227  LPC_EMC->DYNAMICCONTROL = 0x00000000; /* Issue NORMAL command */
228 
229  //[re]enable buffers
230  LPC_EMC->DYNAMICCONFIG0 |= 1<<19;
231 
232  return TRUE;
233 }

235 uint32_t sdram_test( void )
236 {
237  volatile uint32_t *wr_ptr;
238  volatile uint16_t *short_wr_ptr;
239  uint32_t data;
240  uint32_t i, j;
241 
242  wr_ptr = (uint32_t *)SDRAM_BASE;
243  short_wr_ptr = (uint16_t *)wr_ptr;
244 
245  /* 16 bit write */
246  for (i = 0; i < SDRAM_SIZE/0x40000; i++)
247  {
248  for (j = 0; j < 0x100; j++)
249  {
250  *short_wr_ptr++ = (i + j) & 0xFFFF;
251  *short_wr_ptr++ = ((i + j) + 1) & 0xFFFF;
252  }
253  }
254 
255  /* Verifying */
256  wr_ptr = (uint32_t *)SDRAM_BASE;
257  for (i = 0; i < SDRAM_SIZE/0x40000; i++)
258  {
259  for (j = 0; j < 0x100; j++)
260  {
261  data = *wr_ptr;
262  if (data != (((((i + j) + 1) & 0xFFFF) << 16) | ((i + j) & 0xFFFF)))
263  {
264  return 0x0;
265  }
266  wr_ptr++;
267  }
268  }
269  return 0x1;
270 }
