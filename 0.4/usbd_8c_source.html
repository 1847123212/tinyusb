148 tusb_error_t usbd_control_request_subtask(uint8_t coreid, tusb_control_request_t const * const p_request)
149 {
150  OSAL_SUBTASK_BEGIN
151 
152  tusb_error_t error;
153  error = TUSB_ERROR_NONE;
154 
155  //------------- Standard Control such as those in enumeration -------------//
156  if( TUSB_REQUEST_RECIPIENT_DEVICE == p_request->bmRequestType_bit.recipient &&
157  TUSB_REQUEST_TYPE_STANDARD == p_request->bmRequestType_bit.type )
158  {
159  if ( TUSB_REQUEST_GET_DESCRIPTOR == p_request->bRequest )
160  {
161  uint8_t const * p_buffer = NULL;
162  uint16_t length = 0;
163 
164  error = get_descriptor(coreid, p_request, &p_buffer, &length);
165 
166  if ( TUSB_ERROR_NONE == error )
167  {
168  dcd_pipe_control_xfer(coreid, (tusb_direction_t) p_request->bmRequestType_bit.direction, (uint8_t*) p_buffer, length, false);
169  }
170  }
171  else if ( TUSB_REQUEST_SET_ADDRESS == p_request->bRequest )
172  {
173  dcd_controller_set_address(coreid, (uint8_t) p_request->wValue);
174  usbd_devices[coreid].state = TUSB_DEVICE_STATE_ADDRESSED;
175  }
176  else if ( TUSB_REQUEST_SET_CONFIGURATION == p_request->bRequest )
177  {
178  usbd_set_configure_received(coreid, (uint8_t) p_request->wValue);
179  }else
180  {
181  error = TUSB_ERROR_DCD_CONTROL_REQUEST_NOT_SUPPORT;
182  }
183  }
184 
185  //------------- Class/Interface Specific Request -------------//
186  else if ( TUSB_REQUEST_RECIPIENT_INTERFACE == p_request->bmRequestType_bit.recipient)
187  {
188  OSAL_VAR uint8_t class_code;
189 
190  class_code = usbd_devices[coreid].interface2class[ u16_low_u8(p_request->wIndex) ];
191 
192  // TODO [Custom] TUSB_CLASS_DIAGNOSTIC, vendor etc ...
193  if ( (class_code > 0) && (class_code < USBD_CLASS_DRIVER_COUNT) &&
194  usbd_class_drivers[class_code].control_request_subtask )
195  {
196  OSAL_SUBTASK_INVOKED_AND_WAIT( usbd_class_drivers[class_code].control_request_subtask(coreid, p_request), error );
197  }else
198  {
199  error = TUSB_ERROR_DCD_CONTROL_REQUEST_NOT_SUPPORT;
200  }
201  }
202 
203  //------------- Endpoint Request -------------//
204  else if ( TUSB_REQUEST_RECIPIENT_ENDPOINT == p_request->bmRequestType_bit.recipient &&
205  TUSB_REQUEST_TYPE_STANDARD == p_request->bmRequestType_bit.type &&
206  TUSB_REQUEST_CLEAR_FEATURE == p_request->bRequest )
207  {
208  dcd_pipe_clear_stall(coreid, u16_low_u8(p_request->wIndex) );
209  } else
210  {
211  error = TUSB_ERROR_DCD_CONTROL_REQUEST_NOT_SUPPORT;
212  }
213 
214  if(TUSB_ERROR_NONE != error)
215  { // Response with Protocol Stall if request is not supported
216  dcd_pipe_control_stall(coreid);
217  // ASSERT(error == TUSB_ERROR_NONE, VOID_RETURN);
218  }else if (p_request->wLength == 0)
219  {
220  dcd_pipe_control_xfer(coreid, (tusb_direction_t) p_request->bmRequestType_bit.direction, NULL, 0, false); // zero length for non-data
221  }
222 
223  OSAL_SUBTASK_END
224 }

334 static tusb_error_t get_descriptor(uint8_t coreid, tusb_control_request_t const * const p_request, uint8_t const ** pp_buffer, uint16_t * p_length)
335 {
336  tusb_std_descriptor_type_t const desc_type = (tusb_std_descriptor_type_t) u16_high_u8(p_request->wValue);
337  uint8_t const desc_index = u16_low_u8( p_request->wValue );
338 
339  uint8_t const * p_data = NULL ;
340 
341  switch(desc_type)
342  {
343  case TUSB_DESC_TYPE_DEVICE:
344  p_data = tusbd_descriptor_pointers.p_device;
345  (*p_length) = sizeof(tusb_descriptor_device_t);
346  break;
347 
348  case TUSB_DESC_TYPE_CONFIGURATION:
349  p_data = tusbd_descriptor_pointers.p_configuration;
350  (*p_length) = ((tusb_descriptor_configuration_t*)tusbd_descriptor_pointers.p_configuration)->wTotalLength;
351  break;
352 
353  case TUSB_DESC_TYPE_STRING:
354  if ( !(desc_index < 100) ) return TUSB_ERROR_DCD_CONTROL_REQUEST_NOT_SUPPORT; // windows sometimes ask for string at index 238 !!!
355 
356  p_data = tusbd_descriptor_pointers.p_string_arr[desc_index];
357  ASSERT( p_data != NULL, TUSB_ERROR_FAILED);
358 
359  (*p_length) = p_data[0]; // first byte of descriptor is its size
360  break;
361 
362  default: return TUSB_ERROR_DCD_CONTROL_REQUEST_NOT_SUPPORT;
363  }
364 
365  (*p_length) = min16_of(p_request->wLength, (*p_length) ); // cannot return more than hosts requires
366  ASSERT( (*p_length) <= TUSB_CFG_DEVICE_ENUM_BUFFER_SIZE, TUSB_ERROR_NOT_ENOUGH_MEMORY);
367 
368  memcpy(usbd_enum_buffer, p_data, (*p_length));
369  (*pp_buffer) = usbd_enum_buffer;
370 
371  return TUSB_ERROR_NONE;
372 }
