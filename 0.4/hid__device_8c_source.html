185 tusb_error_t hidd_control_request_subtask(uint8_t coreid, tusb_control_request_t const * p_request)
186 {
187  uint8_t subclass_idx;
188  for(subclass_idx=0; subclass_idx<HIDD_NUMBER_OF_SUBCLASS; subclass_idx++)
189  {
190  hidd_interface_t * const p_interface = hidd_class_driver[subclass_idx].p_interface;
191  if ( (p_interface != NULL) && (p_request->wIndex == p_interface->interface_number) ) break;
192  }
193 
194  ASSERT(subclass_idx < HIDD_NUMBER_OF_SUBCLASS, TUSB_ERROR_FAILED);
195 
196  hidd_class_driver_t const * const p_driver = &hidd_class_driver[subclass_idx];
197  hidd_interface_t* const p_hid = p_driver->p_interface;
198 
199  //------------- STD Request -------------//
200  if (p_request->bmRequestType_bit.type == TUSB_REQUEST_TYPE_STANDARD)
201  {
202  uint8_t const desc_type = u16_high_u8(p_request->wValue);
203  uint8_t const desc_index = u16_low_u8 (p_request->wValue);
204 
205  (void) desc_index;
206 
207  ASSERT ( p_request->bRequest == TUSB_REQUEST_GET_DESCRIPTOR && desc_type == HID_DESC_TYPE_REPORT,
208  TUSB_ERROR_DCD_CONTROL_REQUEST_NOT_SUPPORT);
209  ASSERT ( p_hid->report_length <= HIDD_BUFFER_SIZE, TUSB_ERROR_NOT_ENOUGH_MEMORY);
210 
211  memcpy(m_hid_buffer, p_hid->p_report_desc, p_hid->report_length); // to allow report descriptor not to be in USBRAM
212  dcd_pipe_control_xfer(coreid, TUSB_DIR_DEV_TO_HOST, m_hid_buffer, p_hid->report_length, false);
213  }
214  //------------- Class Specific Request -------------//
215  else if (p_request->bmRequestType_bit.type == TUSB_REQUEST_TYPE_CLASS)
216  {
217  OSAL_SUBTASK_BEGIN
218 
219  if( (HID_REQUEST_CONTROL_GET_REPORT == p_request->bRequest) && (p_driver->get_report_cb != NULL) )
220  {
221  // wValue = Report Type | Report ID
222  void* p_buffer = NULL;
223 
224  uint16_t actual_length = p_driver->get_report_cb(coreid, (hid_request_report_type_t) u16_high_u8(p_request->wValue),
225  &p_buffer, p_request->wLength);
226  SUBTASK_ASSERT( p_buffer != NULL && actual_length > 0 );
227 
228  dcd_pipe_control_xfer(coreid, (tusb_direction_t) p_request->bmRequestType_bit.direction, p_buffer, actual_length, false);
229  }
230  else if ( (HID_REQUEST_CONTROL_SET_REPORT == p_request->bRequest) && (p_driver->set_report_cb != NULL) )
231  {
232  // return TUSB_ERROR_DCD_CONTROL_REQUEST_NOT_SUPPORT; // TODO test STALL control out endpoint (with mouse+keyboard)
233  // wValue = Report Type | Report ID
234  tusb_error_t error;
235 
236  dcd_pipe_control_xfer(coreid, (tusb_direction_t) p_request->bmRequestType_bit.direction, m_hid_buffer, p_request->wLength, true);
237 
238  osal_semaphore_wait(usbd_control_xfer_sem_hdl, OSAL_TIMEOUT_NORMAL, &error); // wait for control xfer complete
239  SUBTASK_ASSERT_STATUS(error);
240 
241  p_driver->set_report_cb(coreid, (hid_request_report_type_t) u16_high_u8(p_request->wValue),
242  m_hid_buffer, p_request->wLength);
243  }
244  else if (HID_REQUEST_CONTROL_SET_IDLE == p_request->bRequest)
245  {
246  // uint8_t idle_rate = u16_high_u8(p_request->wValue);
247  }else
248  {
249 // HID_REQUEST_CONTROL_GET_IDLE:
250 // HID_REQUEST_CONTROL_GET_PROTOCOL:
251 // HID_REQUEST_CONTROL_SET_PROTOCOL:
252  return TUSB_ERROR_DCD_CONTROL_REQUEST_NOT_SUPPORT;
253  }
254 
255  OSAL_SUBTASK_END
256  }else
257  {
258  return TUSB_ERROR_DCD_CONTROL_REQUEST_NOT_SUPPORT;
259  }
260 
261  return TUSB_ERROR_NONE;

264 tusb_error_t hidd_open(uint8_t coreid, tusb_descriptor_interface_t const * p_interface_desc, uint16_t *p_length)
265 {
266  uint8_t const *p_desc = (uint8_t const *) p_interface_desc;
267 
268  //------------- HID descriptor -------------//
269  p_desc += p_desc[DESCRIPTOR_OFFSET_LENGTH];
270  tusb_hid_descriptor_hid_t const *p_desc_hid = (tusb_hid_descriptor_hid_t const *) p_desc;
271  ASSERT_INT(HID_DESC_TYPE_HID, p_desc_hid->bDescriptorType, TUSB_ERROR_HIDD_DESCRIPTOR_INTERFACE);
272 
273  //------------- Endpoint Descriptor -------------//
274  p_desc += p_desc[DESCRIPTOR_OFFSET_LENGTH];
275  tusb_descriptor_endpoint_t const *p_desc_endpoint = (tusb_descriptor_endpoint_t const *) p_desc;
276  ASSERT_INT(TUSB_DESC_TYPE_ENDPOINT, p_desc_endpoint->bDescriptorType, TUSB_ERROR_HIDD_DESCRIPTOR_INTERFACE);
277 
278  if (p_interface_desc->bInterfaceSubClass == HID_SUBCLASS_BOOT)
279  {
280  switch(p_interface_desc->bInterfaceProtocol)
281  {
282  case HID_PROTOCOL_KEYBOARD:
283  case HID_PROTOCOL_MOUSE:
284  {
285  hidd_class_driver_t const * const p_driver = &hidd_class_driver[p_interface_desc->bInterfaceProtocol];
286  hidd_interface_t * const p_hid = p_driver->p_interface;
287 
288  ASSERT_PTR(p_hid, TUSB_ERROR_FAILED);
289 
290  p_hid->ept_handle = dcd_pipe_open(coreid, p_desc_endpoint, p_interface_desc->bInterfaceClass);
291  ASSERT( endpointhandle_is_valid(p_hid->ept_handle), TUSB_ERROR_DCD_FAILED);
292 
293  p_hid->interface_number = p_interface_desc->bInterfaceNumber;
294  p_hid->p_report_desc = (p_interface_desc->bInterfaceProtocol == HID_PROTOCOL_KEYBOARD) ? tusbd_descriptor_pointers.p_hid_keyboard_report : tusbd_descriptor_pointers.p_hid_mouse_report;
295  p_hid->report_length = p_desc_hid->wReportLength;
296 
297  ASSERT_PTR(p_hid->p_report_desc, TUSB_ERROR_DESCRIPTOR_CORRUPTED);
298  p_driver->mounted_cb(coreid);
299  }
300  break;
301 
302  default: // TODO unknown, unsupported protocol --> skip this interface
303  return TUSB_ERROR_HIDD_DESCRIPTOR_INTERFACE;
304  }
305  *p_length = sizeof(tusb_descriptor_interface_t) + sizeof(tusb_hid_descriptor_hid_t) + sizeof(tusb_descriptor_endpoint_t);
306  }else
307  {
308  // open generic
309  *p_length = 0;
310  return TUSB_ERROR_HIDD_DESCRIPTOR_INTERFACE;
311  }
312  return TUSB_ERROR_NONE;
313 }
