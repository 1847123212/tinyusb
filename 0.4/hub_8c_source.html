158 tusb_error_t hub_open_subtask(uint8_t dev_addr, tusb_descriptor_interface_t const *p_interface_desc, uint16_t *p_length)
159 {
160  tusb_error_t error;
161 
162  OSAL_SUBTASK_BEGIN
163 
164  // not support multiple TT yet
165  if ( p_interface_desc->bInterfaceProtocol > 1 ) return TUSB_ERROR_HUB_FEATURE_NOT_SUPPORTED;
166 
167  //------------- Open Interrupt Status Pipe -------------//
168  tusb_descriptor_endpoint_t const *p_endpoint;
169  p_endpoint = (tusb_descriptor_endpoint_t const *) descriptor_next( (uint8_t const*) p_interface_desc );
170 
171  SUBTASK_ASSERT(TUSB_DESC_TYPE_ENDPOINT == p_endpoint->bDescriptorType);
172  SUBTASK_ASSERT(TUSB_XFER_INTERRUPT == p_endpoint->bmAttributes.xfer);
173 
174  hub_data[dev_addr-1].pipe_status = hcd_pipe_open(dev_addr, p_endpoint, TUSB_CLASS_HUB);
175  SUBTASK_ASSERT( pipehandle_is_valid(hub_data[dev_addr-1].pipe_status) );
176  hub_data[dev_addr-1].interface_number = p_interface_desc->bInterfaceNumber;
177 
178  (*p_length) = sizeof(tusb_descriptor_interface_t) + sizeof(tusb_descriptor_endpoint_t);
179 
180  //------------- Get Hub Descriptor -------------//
181  OSAL_SUBTASK_INVOKED_AND_WAIT(
182  usbh_control_xfer_subtask( dev_addr, bm_request_type(TUSB_DIR_DEV_TO_HOST, TUSB_REQUEST_TYPE_CLASS, TUSB_REQUEST_RECIPIENT_DEVICE),
183  HUB_REQUEST_GET_DESCRIPTOR, 0, 0,
184  sizeof(descriptor_hub_desc_t), hub_enum_buffer ),
185  error
186  );
187  SUBTASK_ASSERT_STATUS(error);
188 
189  // only care about this field in hub descriptor
190  hub_data[dev_addr-1].port_number = ((descriptor_hub_desc_t*) hub_enum_buffer)->bNbrPorts;
191 
192  //------------- Set Port_Power on all ports -------------//
193  static uint8_t i;
194  for(i=1; i <= hub_data[dev_addr-1].port_number; i++)
195  {
196  OSAL_SUBTASK_INVOKED_AND_WAIT(
197  usbh_control_xfer_subtask( dev_addr, bm_request_type(TUSB_DIR_HOST_TO_DEV, TUSB_REQUEST_TYPE_CLASS, TUSB_REQUEST_RECIPIENT_OTHER),
198  HUB_REQUEST_SET_FEATURE, HUB_FEATURE_PORT_POWER, i,
199  0, NULL ),
200  error
201  );
202  }
203 
204  //------------- Queue the initial Status endpoint transfer -------------//
205  SUBTASK_ASSERT_STATUS ( hcd_pipe_xfer(hub_data[dev_addr-1].pipe_status, &hub_data[dev_addr-1].status_change, 1, true) );
206 
207  OSAL_SUBTASK_END
208 }

tusb_descriptor_endpoint_t::bmAttributes
struct tusb_descriptor_endpoint_t::@8 bmAttributes
This field describes the endpoint's attributes when it is configured using the bConfigurationValue. Bits 1..0: Transfer Type - 00 = Control - 01 = Isochronous - 10 = Bulk - 11 = Interrupt If not an isochronous endpoint, bits 5..2 are reserved and must be set to zero. If isochronous, they are defined as follows: Bits 3..2: Synchronization Type - 00 = No Synchronization - 01 = Asynchronous - 10 = Adaptive - 11 = Synchronous Bits 5..4: Usage Type - 00 = Data endpoint - 01 = Feedback endpoint - 10 = Implicit feedback Data endpoint - 11 = Reserved Refer to Chapter 5 of USB 2.0 specification for more information. All other bits are reserved and must be reset to zero. Reserved bits must be ignored by the host.
