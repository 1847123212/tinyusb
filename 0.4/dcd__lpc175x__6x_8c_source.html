326 static tusb_error_t pipe_control_write(void const * buffer, uint16_t length)
327 {
328  uint32_t const * p_write_data = (uint32_t const *) buffer;
329 
330  LPC_USB->USBCtrl = USBCTRL_WRITE_ENABLE_MASK; // logical endpoint = 0
331  LPC_USB->USBTxPLen = length;
332 
333  for (uint16_t count = 0; count < length_byte2dword(length); count++)
334  {
335  LPC_USB->USBTxData = *p_write_data; // NOTE: cortex M3 have no problem with alignment
336  p_write_data++;
337  }
338 
339  LPC_USB->USBCtrl = 0;
340 
341  // select control IN & validate the endpoint
342  sie_write(SIE_CMDCODE_ENDPOINT_SELECT+1, 0, 0);
343  sie_write(SIE_CMDCODE_BUFFER_VALIDATE , 0, 0);
344 
345  return TUSB_ERROR_NONE;

348 static tusb_error_t pipe_control_read(void * buffer, uint16_t length)
349 {
350  LPC_USB->USBCtrl = USBCTRL_READ_ENABLE_MASK; // logical endpoint = 0
351  while ((LPC_USB->USBRxPLen & USBRXPLEN_PACKET_READY_MASK) == 0) {} // TODO blocking, should have timeout
352 
353  uint16_t actual_length = min16_of(length, (uint16_t) (LPC_USB->USBRxPLen & USBRXPLEN_PACKET_LENGTH_MASK) );
354  uint32_t *p_read_data = (uint32_t*) buffer;
355  for( uint16_t count=0; count < length_byte2dword(actual_length); count++)
356  {
357  *p_read_data = LPC_USB->USBRxData;
358  p_read_data++; // increase by 4 ( sizeof(uint32_t) )
359  }
360 
361  LPC_USB->USBCtrl = 0;
362 
363  // select control OUT & clear the endpoint
364  sie_write(SIE_CMDCODE_ENDPOINT_SELECT+0, 0, 0);
365  sie_write(SIE_CMDCODE_BUFFER_CLEAR , 0, 0);
