217 static void endpoint_non_control_isr(uint32_t int_status)
218 {
219  for(uint8_t ep_id = 2; ep_id < DCD_11U_13U_QHD_COUNT; ep_id++ )
220  {
221  if ( BIT_TEST_(int_status, ep_id) )
222  {
223  dcd_11u_13u_qhd_t * const arr_qhd = dcd_data.qhd[ep_id];
224 
225  // when double buffering, the complete buffer is opposed to the current active buffer in EPINUSE
226  uint8_t const buff_idx = LPC_USB->EPINUSE & BIT_(ep_id) ? 0 : 1;
227  uint16_t const xferred_bytes = dcd_data.current_td[ep_id].queued_bytes_in_buff[buff_idx] - arr_qhd[buff_idx].nbytes;
228 
229  dcd_data.current_td[ep_id].xferred_total += xferred_bytes;
230 
231  // there are still data to transfer.
232  if ( (arr_qhd[buff_idx].nbytes == 0) && (dcd_data.current_td[ep_id].remaining_bytes > 0) )
233  { // NOTE although buff_addr_offset has been increased when xfer is completed
234  // but we still need to increase it one more as we are using double buffering.
235  queue_xfer_to_buffer(ep_id, buff_idx, arr_qhd[buff_idx].buff_addr_offset+1, dcd_data.current_td[ep_id].remaining_bytes);
236  }
237  // short packet or (no more byte and both buffers are finished)
238  else if ( (arr_qhd[buff_idx].nbytes > 0) || !arr_qhd[1-buff_idx].active )
239  { // current TD (request) is completed
240  LPC_USB->EPSKIP = BIT_SET_(LPC_USB->EPSKIP, ep_id); // skip other endpoint in case of short-package
241 
242  dcd_data.current_td[ep_id].remaining_bytes = 0;
243 
244  if ( BIT_TEST_(dcd_data.current_ioc, ep_id) )
245  {
246  endpoint_handle_t edpt_hdl =
247  {
248  .coreid = 0,
249  .index = ep_id,
250  .class_code = dcd_data.class_code[ep_id]
251  };
252 
253  dcd_data.current_ioc = BIT_CLR_(dcd_data.current_ioc, edpt_hdl.index);
254 
255  // TODO no way determine if the transfer is failed or not
256  usbd_xfer_isr(edpt_hdl, TUSB_EVENT_XFER_COMPLETE, dcd_data.current_td[ep_id].xferred_total);
257  }
258 
259  //------------- Next TD is available -------------//
260  if ( dcd_data.next_td[ep_id].total_bytes != 0 )
261  {
262  queue_xfer_in_next_td(ep_id);
263  }
264  }else
265  {
266  // transfer complete, there is no more remaining bytes, but this buffer is not the last transaction (the other is)
267  }
268  }
269  }

296 void dcd_isr(uint8_t coreid)
297 {
298  (void) coreid;
299 
300  uint32_t const int_enable = LPC_USB->INTEN;
301  uint32_t const int_status = LPC_USB->INTSTAT & int_enable;
302  LPC_USB->INTSTAT = int_status; // Acknowledge handled interrupt
303 
304  if (int_status == 0) return;
305 
306  uint32_t const dev_cmd_stat = LPC_USB->DEVCMDSTAT;
307 
308  //------------- Device Status -------------//
309  if ( int_status & INT_MASK_DEVICE_STATUS )
310  {
311  LPC_USB->DEVCMDSTAT |= CMDSTAT_RESET_CHANGE_MASK | CMDSTAT_CONNECT_CHANGE_MASK | CMDSTAT_SUSPEND_CHANGE_MASK;
312  if ( dev_cmd_stat & CMDSTAT_RESET_CHANGE_MASK) // bus reset
313  {
314  bus_reset();
315  usbd_dcd_bus_event_isr(0, USBD_BUS_EVENT_RESET);
316  }
317 
318  if (dev_cmd_stat & CMDSTAT_CONNECT_CHANGE_MASK)
319  { // device disconnect
320  if (dev_cmd_stat & CMDSTAT_DEVICE_ADDR_MASK)
321  { // debouncing as this can be set when device is powering
322  usbd_dcd_bus_event_isr(0, USBD_BUS_EVENT_UNPLUGGED);
323  }
324  }
325 
326  // TODO support suspend & resume
327  if (dev_cmd_stat & CMDSTAT_SUSPEND_CHANGE_MASK)
328  {
329  if (dev_cmd_stat & CMDSTAT_DEVICE_SUSPEND_MASK)
330  { // suspend signal, bus idle for more than 3ms
331  // Note: Host may delay more than 3 ms before and/or after bus reset before doing enumeration.
332  if (dev_cmd_stat & CMDSTAT_DEVICE_ADDR_MASK)
333  {
334  usbd_dcd_bus_event_isr(0, USBD_BUS_EVENT_SUSPENDED);
335  }
336  }
337  }
338 // else
339 // { // resume signal
340 // usbd_dcd_bus_event_isr(0, USBD_BUS_EVENT_RESUME);
341 // }
342 // }
343  }
344 
345  //------------- Setup Received -------------//
346  if ( BIT_TEST_(int_status, 0) && (dev_cmd_stat & CMDSTAT_SETUP_RECEIVED_MASK) )
347  { // received control request from host
348  // copy setup request & acknowledge so that the next setup can be received by hw
349  usbd_setup_received_isr(coreid, &dcd_data.setup_request);
350 
351  // NXP control flowchart clear Active & Stall on both Control IN/OUT endpoints
352  dcd_data.qhd[0][0].stall = dcd_data.qhd[1][0].stall = 0;
353 
354  LPC_USB->DEVCMDSTAT |= CMDSTAT_SETUP_RECEIVED_MASK;
355  dcd_data.qhd[0][1].buff_addr_offset = addr_offset(&dcd_data.setup_request);
356  }
357  //------------- Control Endpoint -------------//
358  else if ( int_status & 0x03 )
359  {
360  endpoint_control_isr(int_status);
361  }
362 
363  //------------- Non-Control Endpoints -------------//
364  if( int_status & ~(0x03UL) )
365  {
366  endpoint_non_control_isr(int_status);
367  }
368 }

514  dcd_data.current_td[ep_id].xferred_total = 0;

This field describes the endpoint's attributes when it is configured using the bConfigurationValue. Bits 1..0: Transfer Type - 00 = Control - 01 = Isochronous - 10 = Bulk - 11 = Interrupt If not an isochronous endpoint, bits 5..2 are reserved and must be set to zero. If isochronous, they are defined as follows: Bits 3..2: Synchronization Type - 00 = No Synchronization - 01 = Asynchronous - 10 = Adaptive - 11 = Synchronous Bits 5..4: Usage Type - 00 = Data endpoint - 01 = Feedback endpoint - 10 = Implicit feedback Data endpoint - 11 = Reserved Refer to Chapter 5 of USB 2.0 specification for more information. All other bits are reserved and must be reset to zero. Reserved bits must be ignored by the host.
